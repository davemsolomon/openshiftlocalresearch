### catalogsource.yml 
  apiVersion: operators.coreos.com/v1alpha1
  kind: CatalogSource
  metadata:
    name: operatorhubio-catalog
    namespace: olm
  spec:
    sourceType: grpc
    image: quay.io/operatorhubio/catalog:latest
    displayName: OperatorHub Community Operators
    publisher: OperatorHub.io

oc apply -f ./catalogsource.yml
